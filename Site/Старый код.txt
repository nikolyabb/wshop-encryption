    <!-- Добавляем JavaScript код для карты -->
    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const image = document.getElementById('clickable-image');
            const map = document.querySelector('map[name="image-map"]');
            
            // Области для клика (координаты нужно подобрать под ваше изображение)
            const areas = [
                { 
                    name: 'ASGARD',
                    coords: [100, 50, 200, 150], // x1,y1,x2,y2
                    url: '/asgard.html'
                },
                { 
                    name: 'MIDGARD',
                    coords: [300, 50, 100, 150],
                    url: 'midgard.html'
                },
                { 
                    name: 'MUSPELHEIM',
                    coords: [462, 100, 100, 200],
                    url: '/muspelheim.html'
                },
                { 
                    name: 'VANAHEIM',
                    coords: [150, 300, 250, 400],
                    url: '/vanaheim.html'
                },
                { 
                    name: 'HELHEIM',
                    coords: [400, 400, 500, 500],
                    url: '/helheim.html'
                }
            ];

            // Создаем области на карте
            areas.forEach(area => {
                const areaElement = document.createElement('area');
                areaElement.shape = 'rect';
                areaElement.coords = area.coords.join(',');
                areaElement.alt = area.name;
                areaElement.href = area.url;
                map.appendChild(areaElement);
            });
            
            console.log('Карта миров Yggdrasil активирована!');

            image.addEventListener('click', function(event) {
    // 1. Получаем положение изображения на странице
    const rect = image.getBoundingClientRect();
    
    // 2. Вычисляем координаты клика ОТНОСИТЕЛЬНО изображения
    const x = Math.round(event.clientX - rect.left);
    const y = Math.round(event.clientY - rect.top);
    
    // 3. Показываем результат
    console.log(`Координаты: x=${x}, y=${y}`);
    
    // 4. Можно вывести alert для удобства
    alert(`Клик: x=${x}, y=${y}\n\nСкопируйте эти числа в массив areas:\n[x, y, x+ширина, y+высота]`);
    
    // 5. Визуальная обратная связь (опционально)
    const dot = document.createElement('div');
    dot.style.position = 'absolute';
    dot.style.left = (event.clientX - 5) + 'px';
    dot.style.top = (event.clientY - 5) + 'px';
    dot.style.width = '10px';
    dot.style.height = '10px';
    dot.style.backgroundColor = 'red';
    dot.style.borderRadius = '50%';
    dot.style.pointerEvents = 'none';
    document.body.appendChild(dot);
    
    // Автоудаление точки через 2 секунды
    setTimeout(() => dot.remove(), 2000);
});
            
            // Дополнительный код для отладки (узнать координаты кликов)
